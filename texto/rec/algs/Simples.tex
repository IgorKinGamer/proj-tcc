% Simples (ascendente)
\begin{algorithm}
	\caption{\Simples\ -- \fACMP\ subindo as ligações}
	\label{alg:simples}
	
	\Entrada{Dois nós $a$ e $b$}
	\Saida{O ACMP entre $a$ e $b$}
	\BlankLine
	
	\tcp{Inicialmente, encontra ancestrais de $a$ e $b$ no mesmo nível}
	\Enqto {$\var{nível}_a > \var{nível}_b$}
		{$a \gets \pai(a)$}
	\Enqto {$\var{nível}_b > \var{nível}_a$}
		{$b \gets \pai(b)$}
		
	\BlankLine
	\Enqto {$a \not= b$}
	{
		$a \gets \pai(a)$ \\
		$b \gets \pai(b)$ \\
	}
	\Retorna {$a$}
\end{algorithm}
